300void RecordProjector::adjust_assumptions_simple(int days)
301{
302 double duration_factor = days / 360.0;
303 double sum_row_nondiag;
304
305 // simple scaling method
306 for (unsigned r = 0; r < _dimension; r++)
307 {
308 sum_row_nondiag = 0;
309 for (unsigned c = 0; c < _dimension; c++)
310 {
311 if (c == r)
312 {
313 continue;
314 }
315 double this_scaled_val = duration_factor * be_a_yearly.get()[r * _dimension + c];
316 sum_row_nondiag += this_scaled_val;
317 be_a_time_step_dependent.get()[r * _dimension + c] = this_scaled_val;
318 }
319 be_a_time_step_dependent.get()[r * _dimension + r] = 1 - sum_row_nondiag;
320 }
