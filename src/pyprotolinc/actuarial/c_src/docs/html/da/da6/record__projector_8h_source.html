400void RecordProjector::adjust_assumptions_simple(int days)
401{
402 double duration_factor = days / 360.0;
403 double sum_row_nondiag;
404
405 // simple scaling method
406 for (unsigned r = 0; r < _dimension; r++)
407 {
408 sum_row_nondiag = 0;
409 for (unsigned c = 0; c < _dimension; c++)
410 {
411 if (c == r)
412 {
413 continue;
414 }
415 double this_scaled_val = duration_factor * be_a_yearly.get()[r * _dimension + c];
416 sum_row_nondiag += this_scaled_val;
417 be_a_time_step_dependent.get()[r * _dimension + c] = this_scaled_val;
418 }
419 be_a_time_step_dependent.get()[r * _dimension + r] = 1 - sum_row_nondiag;
420 }
