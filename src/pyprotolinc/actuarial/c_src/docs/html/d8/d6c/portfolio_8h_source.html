409int get_age_at_date(int dob_year, int dob_month, int dob_day, int dt_year, int dt_month, int dt_day)
410{
411 // check that person is born at dt already
412 if (dt_year < dob_year || (dt_year == dob_year && dt_month < dob_month) ||
413 (dt_year == dob_year && dt_month == dob_month && dt_day < dob_day))
414 {
415 return -1; // use as a error signal
416 }
417
418 // special case equality of dates
419 if (dt_year == dob_year && dt_month == dob_month && dt_day == dob_day)
420 {
421 return 0;
422 }
423
424 int full_years = (dt_month > dob_month || dob_month == dt_month && dob_day <= dt_day) ? dt_year - dob_year : dt_year - dob_year - 1;
425
426 // so the last birthday before dt was at dob_year + full_years/dob_month/dob_day
427 int full_months = (dt_month < dob_month || (dt_month == dob_month && dt_day < dob_day)) ? 12 + dt_month - dob_month - (dt_day < dob_day ? 1 : 0)
428 : dt_month - dob_month - (dt_day < dob_day ? 1 : 0);
429
430 return 12 * full_years + full_months;
431}
