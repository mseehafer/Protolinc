100int MetaRunner::get_num_groups()
101{
102 if (!run_config.get_use_multicore())
103 {
104 return 1;
105 }
106 int cpu_count = run_config.get_cpu_count();
107 int ptf_size = (int) ptr_portfolio->size();
108
109 // make sure we have at least four policies in each group
110 int tmp = cpu_count < ptf_size / 4 ? cpu_count : ptf_size / 4;
111 return tmp == 0 ? 1 : tmp;

114// calculate the result and store in ext_result
115void MetaRunner::run(RunResult &run_result)
116{
117 cout << "MetaRunner::run(): STARTING RUN" << endl;
118 const CAssumptionSet &be_ass = run_config.get_be_assumptions();
119 unsigned dimension = be_ass.get_dimension();
120
121 cout << "MetaRunner::run(): dimension=" << be_ass.get_dimension() << endl;
122 cout << "MetaRunner::run(): portfolio.size()=" << ptr_portfolio->size() << endl;
123
124 for (unsigned r = 0; r < dimension; r++)
125 {
126 for (unsigned c = 0; c < dimension; c++)
127 {
128 cout << "MetaRunner::run(): (" << r << ", " << c << "): " << be_ass.get_provider_info(r, c) << endl;
129 }
130 }
131
132 // create N runners, run_results and sub-portfolios
133 const int NUM_GROUPS = get_num_groups();
134 cout << "MetaRunner::run(): NUM_GROUPS=" << NUM_GROUPS << endl;
135 vector<shared_ptr<CPolicyPortfolio>> subportfolios(NUM_GROUPS);
136 vector<Runner> runners = vector<Runner>();
137 vector<RunResult> results = vector<RunResult>();
138 for (int j = 0; j < NUM_GROUPS; j++)
139 {
140 subportfolios[j] = make_shared<CPolicyPortfolio>(ptr_portfolio->_ptf_year, ptr_portfolio->_ptf_month, ptr_portfolio->_ptf_day);
141 runners.emplace_back(Runner(j + 1, subportfolios[j], run_config, _ta));
142 results.emplace_back(RunResult(_ta));
143 }
144
145 // split portfolio into N groups
146 int subportfolio_index = 0;
147 const vector<shared_ptr<CPolicy>> &policies = ptr_portfolio->get_policies();
148 for (shared_ptr<CPolicy> record : policies)
149 {
150 subportfolios[subportfolio_index]->add(record);
151 if (++subportfolio_index >= NUM_GROUPS)
152 {
153 subportfolio_index = 0;
154 }
155 }
156
157 // value subportfolios
158#pragma omp parallel for
159 for (int j = 0; j < NUM_GROUPS; j++)
160 {
161 runners[j].run(results[j]);
162 }
163
164 // combine the results of the subportfolios to combined result
165 for (int j = 0; j < NUM_GROUPS; j++)
166 {
167 run_result.add_result(results[j]);
168 }
169
170 cout << "MetaRunner::run(): DONE" << endl;
171}
