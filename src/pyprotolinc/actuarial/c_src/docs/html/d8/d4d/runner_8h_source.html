153void MetaRunner::run(RunResult &run_result)
154{
155 cout << "MetaRunner::run(): STARTING RUN" << endl;
156 const CAssumptionSet &be_ass = _run_config.get_be_assumptions();
157 unsigned dimension = be_ass.get_dimension();
158
159 cout << "MetaRunner::run(): dimension=" << be_ass.get_dimension() << endl;
160 cout << "MetaRunner::run(): portfolio.size()=" << _ptr_portfolio->size() << endl;
161
162 for (unsigned r = 0; r < dimension; r++)
163 {
164 for (unsigned c = 0; c < dimension; c++)
165 {
166 cout << "MetaRunner::run(): (" << r << ", " << c << "): " << be_ass.get_provider_info(r, c) << endl;
167 }
168 }
169
170 // create N runners, run_results and sub-portfolios
171 const int NUM_GROUPS = get_num_groups();
172 cout << "MetaRunner::run(): NUM_GROUPS=" << NUM_GROUPS << endl;
173 vector<shared_ptr<CPolicyPortfolio>> subportfolios(NUM_GROUPS);
174 vector<Runner> runners = vector<Runner>();
175 vector<RunResult> results = vector<RunResult>();
176 for (int j = 0; j < NUM_GROUPS; j++)
177 {
178 //subportfolios[j] = make_shared<CPolicyPortfolio>(_ptr_portfolio->_ptf_year, _ptr_portfolio->_ptf_month, _ptr_portfolio->_ptf_day);
179 subportfolios[j] = make_shared<CPolicyPortfolio>(_ptr_portfolio->get_portfolio_date());
180 runners.emplace_back(Runner(j + 1, subportfolios[j], _run_config, _ta));
181 results.emplace_back(RunResult(_ta));
182 }
183
184 // split portfolio into N groups
185 int subportfolio_index = 0;
186 const vector<shared_ptr<CPolicy>> &policies = _ptr_portfolio->get_policies();
187 for (shared_ptr<CPolicy> record : policies)
188 {
189 subportfolios[subportfolio_index]->add(record);
190 if (++subportfolio_index >= NUM_GROUPS)
191 {
192 subportfolio_index = 0;
193 }
194 }
195
196 // value subportfolios
197#pragma omp parallel for
198 for (int j = 0; j < NUM_GROUPS; j++)
199 {
200 runners[j].run(results[j]);
201 }
202
203 // combine the results of the subportfolios to combined result
204 for (int j = 0; j < NUM_GROUPS; j++)
205 {
206 run_result.add_result(results[j]);
207 }
208
209 cout << "MetaRunner::run(): DONE" << endl;
210}
